# Required 
- Required: node v7.4.x
- npm i -g nodemon
- create a VHOST my_back.dev

# Install and run (host mode)
- cd api
- npm install
- cd ..
- npm run start

# Install and run (docker mode: nodejs + mysql)
- cd api
- npm install
- cd ..
- npm run docker-start
